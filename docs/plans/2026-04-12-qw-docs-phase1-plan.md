**Step 3: Implement `compute_optical_matrix_qw` in `gfactor_functions.f90`**

Add after `compute_optical_matrix_wire` (after line 1132):

```fortran
subroutine compute_optical_matrix_qw(transitions, num_trans, &
  & cb_state, vb_state, cb_value, vb_value, numcb, numvb, &
  & profile, kpterms, cfg)
  use defs
  implicit none
  type(optical_transition), allocatable, intent(out) :: transitions(:)
  integer, intent(out) :: num_trans
  real(kind=dp), intent(in) :: cb_state(:,:), vb_state(:,:)
  real(kind=dp), intent(in) :: cb_value(:), vb_value(:)
  integer, intent(in) :: numcb, numvb
  real(kind=dp), intent(in) :: profile(:,:), kpterms(:,:,:)
  type(simulation_config), intent(in) :: cfg
  integer :: i, j, k, d, N, idx
  real(kind=dp) :: Pele(3), dE

  N = size(cb_state, 1)
  allocate(transitions(numcb * numvb))
  num_trans = 0

  do i = 1, numcb
    do j = 1, numvb
      dE = cb_value(i) - vb_value(j)
      if (dE <= 0.0_dp) cycle

      num_trans = num_trans + 1
      transitions(num_trans)%cb_idx = i
      transitions(num_trans)%vb_idx = j
      transitions(num_trans)%energy = dE

      ! Compute momentum matrix elements in x, y, z
      do d = 1, 3
        call pMatrixEleCalc(Pele(d), d, &
          & cb_state(:,i), vb_state(:,j), &
          & cfg%numLayers, params_for_optics, &
          & .false., profile, kpterms, &
          & HT_csr=dummy_csr, hkp_out=dummy_hkp, &
          & startz=cfg%startz, endz=cfg%endz, dz=cfg%dz)
      end do

      transitions(num_trans)%px = abs(Pele(1))**2
      transitions(num_trans)%py = abs(Pele(2))**2
      transitions(num_trans)%pz = abs(Pele(3))**2
      transitions(num_trans)%oscillator_strength = &
        & (abs(Pele(1))**2 + abs(Pele(2))**2 + abs(Pele(3))**2) &
        & / (hbar2O2m0 * dE)
    end do
  end do

  ! Trim to actual count
  transitions = transitions(1:num_trans)
end subroutine
```

Note: The exact signature needs to match the existing `pMatrixEleCalc` calling convention from gfactor_functions.f90:192. The implementer must read `pMatrixEleCalc` carefully and match its parameter list.
